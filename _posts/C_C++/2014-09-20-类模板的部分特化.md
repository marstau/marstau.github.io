```
 #include "stdafx.h"

template<class T, class Q>
class C{
public:
    C( T t, Q q )
        :_t(t), _q(q){}
    void print(){ cout << _t << " " << _q << endl; }
private:
    T _t;
    Q _q;
};

template<class T>
class C<T, int>{
public:
    C( T t, int q )
        :_t(t),_q(q*2){}
    void print(){ cout << _t << " " << _q << endl; }
private:
    T _t;
    int _q;
};
int main(){
    C<float, float> c1(1.0f,3.0f);
    c1.print();
    C<float, int> c2(1.1f,3);
    c2.print();
}
/*
1 3
1.1 6
请按任意键继续. . .
*/
```
